# 브랜드별 자동 분류 쿼리
UPDATE TG_SL_StockList_Temp SET
SL_Code = (CASE
  WHEN substring(`JANコード`,1,4) = 'EVLW' THEN 'envylook'
  WHEN substring(`JANコード`,1,4) = 'CAFW' AND substring(`商品名`,1,11) = '[MONODAILY]'
    THEN 'sweets24'
  WHEN substring(`JANコード`,1,4) = 'LRHW' THEN 'laurenhi'
  ELSE SL_Code END);
